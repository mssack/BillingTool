CREATE TABLE [CashBook]
(
	[Id] uniqueidentifier PRIMARY KEY DEFAULT newid(),
	[ReferenceNumber] integer NOT NULL UNIQUE,
	[Date] datetime NOT NULL DEFAULT getdate(),
	[AmountGross] money NOT NULL DEFAULT(0),
	[TaxPercent] money NOT NULL DEFAULT(0),
	[Issuer] ntext NOT NULL,
	[Recipient] ntext NOT NULL,
	[Text] ntext,


	[InternRecipientId] ntext,
	[InternDescription] ntext,
	[LastEdited] datetime NOT NULL DEFAULT getdate()
)
